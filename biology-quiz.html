        // Generate quiz questions based on selected topics

        function generateQuizQuestions() {

            // Filter questions by selected topics

            let availableQuestions = questionsDatabase.filter(q => {

            availableQuestions = shuffleArray(availableQuestions);

            

            // Take the requested number of questions

                numberOfQuestions = quizQuestions.length;

                updateQuestionCount();

            }

            

            // Initialize user answers array

            userAnswers = new Array(quizQuestions.length).fill(-1);

        }


        // Shuffle array using Fisher-Yates algorithm

        function shuffleArray(array) {

            const newArray = [...array];

            for (let i = newArray.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [newArray[i], newArray[j]] = [newArray[j], newArray[i]];

            }

            return newArray;

        }

            userAnswers[currentQuestionIndex] = optionIndex;

            

            // Update UI to show selected option

            const options = document.querySelectorAll('.option');

            options.forEach((option, i) => {

                option.classList.toggle('selected', i === optionIndex);

            });

            

            // Update navigation buttons

            updateNavigationButtons();

            

            // Update score display

            updateScoreDisplay();

        }


        // Update navigation buttons state

        function updateNavigationButtons() {

            // Previous button

            prevBtn.style.display = currentQuestionIndex > 0 ? 'inline-flex' : 'none';

            

            // Next/Submit button

            if (currentQuestionIndex < quizQuestions.length - 1) {

                nextBtn.style.display = 'inline-flex';

                submitBtn.style.display = 'none';

            } else {

                nextBtn.style.display = 'none';

                submitBtn.style.display = 'inline-flex';

            }

            

            // Enable/disable next button based on whether an answer is selected

            nextBtn.disabled = userAnswers[currentQuestionIndex] === -1;

            submitBtn.disabled = userAnswers[currentQuestionIndex] === -1;

        }


        // Navigate to next question

        function nextQuestion() {

            if (currentQuestionIndex < quizQuestions.length - 1) {

                currentQuestionIndex++;

                displayQuestion(currentQuestionIndex);

                updateNavigationButtons();

            }

        }


        // Navigate to previous question

        function previousQuestion() {

            if (currentQuestionIndex > 0) {

                currentQuestionIndex--;

                displayQuestion(currentQuestionIndex);

                updateNavigationButtons();

            }

        }


        // Submit the quiz

        function submitQuiz() {

            // Stop timer

            clearInterval(timerInterval);

            

            // Calculate score

            let correctCount = 0;

            quizQuestions.forEach((question, index) => {

                if (userAnswers[index] === question.correct) {

                    correctCount++;

                }

            });

            

            // Calculate percentage

            const percentage = Math.round((correctCount / quizQuestions.length) * 100);

            

            // Switch to results screen

            questionScreen.style.display = 'none';

            resultsScreen.style.display = 'block';

            document.getElementById('quizTitle').textContent = 'Biology Quiz - Results';

            

            // Update results display

            scorePercent.textContent = `${percentage}%`;

            correctAnswers.textContent = correctCount;

            totalQuizQuestions.textContent = quizQuestions.length;

            

            // Format time taken

            const minutes = Math.floor(timeElapsed / 60);

            const seconds = timeElapsed % 60;

            timeTaken.textContent = `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;

            

            // Update score circle with animation

            const scoreCircleElement = document.getElementById('scoreCircle');

            scoreCircleElement.style.background = `conic-gradient(var(--primary-green) 0% ${percentage}%, var(--light-green) ${percentage}% 100%)`;

            

            // Generate results breakdown

            generateResultsBreakdown(correctCount);

        }


        // Generate results breakdown

        function generateResultsBreakdown(correctCount) {

            breakdownList.innerHTML = '';

            

            quizQuestions.forEach((question, index) => {

                const isCorrect = userAnswers[index] === question.correct;

                const breakdownItem = document.createElement('div');

                breakdownItem.className = 'breakdown-item';

                

                breakdownItem.innerHTML = `

                    <div class="breakdown-question">

                        <strong>Q${index + 1}:</strong> ${question.text.substring(0, 80)}...

                        <div style="font-size: 0.85rem; color: #666; margin-top: 5px;">

                            Topic: ${question.topic} | Year: ${question.year} | Difficulty: ${question.difficulty}

                        </div>

                    </div>

                    <div class="breakdown-status ${isCorrect ? 'correct' : 'incorrect'}">

                        ${isCorrect ? 'Correct' : 'Incorrect'}

                    </div>

                `;

                

                breakdownList.appendChild(breakdownItem);

            });

        }


        // Update timer display

        function updateTimer() {

            timeElapsed++;

            const minutes = Math.floor(timeElapsed / 60);

            const seconds = timeElapsed % 60;

            timerElement.textContent = `Time: ${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;

        }


        // Update score display

        function updateScoreDisplay() {

            let answeredCount = 0;

            let correctCount = 0;

            

            for (let i = 0; i <= currentQuestionIndex; i++) {

                if (userAnswers[i] !== -1) {

                    answeredCount++;

                    if (userAnswers[i] === quizQuestions[i].correct) {

                        correctCount++;

                    }

                }

            }

            

            scoreDisplay.textContent = `Score: ${correctCount}/${answeredCount}`;

        }


        // Restart quiz

        function restartQuiz() {

            // Reset UI

            resultsScreen.style.display = 'none';

            welcomeScreen.style.display = 'block';

            document.getElementById('quizTitle').textContent = 'Biology Quiz';

            

            // Reset quiz state

            currentQuestionIndex = 0;

            userAnswers = [];

            quizQuestions = [];

            quizStarted = false;

            timeElapsed = 0;

            timerElement.textContent = 'Time: 00:00';

            scoreDisplay.textContent = 'Score: 0/0';

            

            // Clear timer

            clearInterval(timerInterval);

        }


        // Initialize the app

        function init() {

            // Initialize topics

            initializeTopics();

            

            // Set up slider

            questionSlider.addEventListener('input', () => {

                numberOfQuestions = parseInt(questionSlider.value);

                updateQuestionCount();

            });

            

            // Set up buttons

            selectAllBtn.addEventListener('click', () => {

                const allSelected = biologyTopics.every(topic => topic.selected);

                biologyTopics.forEach(topic => {

                    topic.selected = !allSelected;

                });

                initializeTopics();

            });

            

            clearAllBtn.addEventListener('click', () => {

                biologyTopics.forEach(topic => {

                    topic.selected = false;

                });

                initializeTopics();

            });

            

            startQuizBtn.addEventListener('click', startQuiz);

            prevBtn.addEventListener('click', previousQuestion);

            nextBtn.addEventListener('click', nextQuestion);

            submitBtn.addEventListener('click', submitQuiz);

            restartBtn.addEventListener('click', restartQuiz);

            reviewBtn.addEventListener('click', () => {

                // Go back to question screen to review answers

                resultsScreen.style.display = 'none';

                questionScreen.style.display = 'block';

                document.getElementById('quizTitle').textContent = 'Biology Quiz - Review';

                

                // Show all answers

                displayQuestion(0);

                updateNavigationButtons();

                

                // Disable option selection during review

                const options = document.querySelectorAll('.option');

                options.forEach(option => {

                    option.style.pointerEvents = 'none';

                });

                

                // Change button text

                nextBtn.textContent = 'Review Complete';

                nextBtn.addEventListener('click', () => {

                    // Go back to results

                    questionScreen.style.display = 'none';

                    resultsScreen.style.display = 'block';

                }, { once: true });

            });

            

            // Initialize question count display

            updateQuestionCount();

        }


        // Start the app when page loads

        document.addEventListener('DOMContentLoaded', init);

    </script>
